ready

Client

Emitted when the client becomes ready to start working.

Properties

This event provides the User object of the bot.

// src/events/client/Ready.ts
export class ReadyEvent extends Event {
    name = "ready";
    once = true;

    execute(client, data) {
        Logger.success(`Logged in as key: ${data.user.username}`);
    }
}
